# Footex Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a browser-based FIFA World Cup 2026 fan app that displays group standings, lets users update match results inline, and predicts match outcomes using team stats.

**Architecture:** Pure HTML + CSS + ES Modules split into four layers: static data (`data/`), a pure prediction engine (`engine/`), UI components and app logic (`ui/`), and a single HTML entry point. State (match results) lives in `localStorage`; all team data is static in JS files.

**Tech Stack:** HTML5, CSS custom properties, vanilla ES Modules (no bundler, no framework). Served locally via `npx serve .` or `python -m http.server`.

---

## File Map

| File | Responsibility |
|---|---|
| `index.html` | Entry point: nav bar, view containers, `<script type="module">` |
| `style.css` | Pitch Green theme via CSS custom properties, all layout |
| `data/teams.js` | Export `TEAMS` — array of 32 team objects |
| `data/schedule.js` | Export `SCHEDULE` — array of fixture objects with matchId |
| `engine/predictor.js` | Export `predict(teamA, teamB, allTeams)` → prediction object |
| `ui/components.js` | Export pure functions that return HTML strings |
| `ui/app.js` | Import everything, manage state, render views, wire events |

// engine/predictor.js

function normalize(value, min, max) {
  if (max === min) return 0;
  return (value - min) / (max - min);
}

export function teamStrength(team, allTeams) {
  const rankings = allTeams.map(t => t.fifaRanking);
  const worstRank = Math.max(...rankings);
  const bestRank = Math.min(...rankings);
  // FIFA: rank 1 (best) → 100, worst rank in pool → 0
  const rankScore = normalize(worstRank - team.fifaRanking, 0, worstRank - bestRank);

  const formScore = team.formPoints / 30; // max 30 = 10 wins

  const maxGpg = Math.max(...allTeams.map(t => t.goalsPerGame));
  const minGpg = Math.min(...allTeams.map(t => t.goalsPerGame));
  const attackScore = normalize(team.goalsPerGame, minGpg, maxGpg);

  const maxConceded = Math.max(...allTeams.map(t => t.goalsAgainstPerGame));
  const minConceded = Math.min(...allTeams.map(t => t.goalsAgainstPerGame));
  // Inverted: lower goals against = better defense
  const defenseScore = normalize(maxConceded - team.goalsAgainstPerGame, 0, maxConceded - minConceded);

  const maxExp = Math.max(...allTeams.map(t => t.worldCupAppearances + t.starPlayers * 2));
  const expRaw = team.worldCupAppearances + team.starPlayers * 2;
  const expScore = normalize(expRaw, 0, maxExp);

  const strength =
    rankScore * 0.25 +
    formScore * 0.25 +
    attackScore * 0.20 +
    defenseScore * 0.20 +
    expScore * 0.10;

  return Math.round(strength * 100);
}

export function predict(teamA, teamB, allTeams) {
  const strengthA = teamStrength(teamA, allTeams);
  const strengthB = teamStrength(teamB, allTeams);
  const diff = strengthA - strengthB; // positive = A stronger

  // Convert strength diff to win probabilities using a logistic-style spread
  // diff of 0 → 45% / 22% / 33% (slight home advantage skew toward draw)
  // diff of 50 → ~75% / 15% / 10%
  const rawA = 0.45 + diff * 0.006;
  const rawDraw = 0.22 - Math.abs(diff) * 0.002;
  const rawB = 1 - rawA - rawDraw;

  const clamp = (v, lo, hi) => Math.max(lo, Math.min(hi, v));
  const pA = clamp(rawA, 0.05, 0.90);
  const pDraw = clamp(rawDraw, 0.05, 0.35);
  const pB = clamp(1 - pA - pDraw, 0.05, 0.90);

  // Normalise so they sum to 1
  const total = pA + pDraw + pB;
  const homeWinPct = Math.round((pA / total) * 100);
  const drawPct = Math.round((pDraw / total) * 100);
  const awayWinPct = 100 - homeWinPct - drawPct;

  // Expected goals: attack score × (1 - opponent defense ratio)
  const maxGpg = Math.max(...allTeams.map(t => t.goalsPerGame));
  const defRatioB = teamB.goalsAgainstPerGame / maxGpg;
  const defRatioA = teamA.goalsAgainstPerGame / maxGpg;
  const xgA = Math.max(0, teamA.goalsPerGame * (1 - defRatioB * 0.5));
  const xgB = Math.max(0, teamB.goalsPerGame * (1 - defRatioA * 0.5));

  const scoreA = Math.round(xgA);
  const scoreB = Math.round(xgB);

  const favorite = strengthA >= strengthB ? teamA.id : teamB.id;
  const underdog = strengthA >= strengthB ? teamB.id : teamA.id;

  return {
    homeWinPct,
    drawPct,
    awayWinPct,
    predictedScore: { home: scoreA, away: scoreB },
    favorite,
    favoriteStrength: Math.max(strengthA, strengthB),
    underdog,
    underdogStrength: Math.min(strengthA, strengthB),
    strengthA,
    strengthB,
  };
}

export function groupTable(group, teams, fixtures, results) {
  const groupTeams = teams.filter(t => t.group === group);
  const groupFixtures = fixtures.filter(f => f.group === group);

  // Calculate standings
  const standing = {};
  groupTeams.forEach(t => {
    standing[t.id] = { id: t.id, p: 0, w: 0, d: 0, l: 0, gf: 0, ga: 0, pts: 0 };
  });

  groupFixtures.forEach(f => {
    const r = results[f.id];
    if (!r) return;
    const h = standing[f.home];
    const a = standing[f.away];
    if (!h || !a) return;
    h.p++; a.p++;
    h.gf += r.home; h.ga += r.away;
    a.gf += r.away; a.ga += r.home;
    if (r.home > r.away) { h.w++; h.pts += 3; a.l++; }
    else if (r.home < r.away) { a.w++; a.pts += 3; h.l++; }
    else { h.d++; h.pts++; a.d++; a.pts++; }
  });

  const sorted = Object.values(standing).sort((a, b) =>
    b.pts - a.pts || (b.gf - b.ga) - (a.gf - a.ga) || b.gf - a.gf
  );

  const teamById = Object.fromEntries(teams.map(t => [t.id, t]));

  const rows = sorted.map(s => {
    const t = teamById[s.id];
    return `
      <tr>
        <td><button class="team-name-btn" data-team="${t.id}">${t.flag} ${t.name}</button></td>
        <td>${s.p}</td><td>${s.w}</td><td>${s.d}</td><td>${s.l}</td>
        <td>${s.gf}</td><td>${s.ga}</td><td>${s.gf - s.ga}</td>
        <td class="pts">${s.pts}</td>
      </tr>`;
  }).join('');

  const fixtureRows = groupFixtures.map(f => {
    const r = results[f.id];
    const homeTeam = teamById[f.home];
    const awayTeam = teamById[f.away];
    const scoreHtml = r
      ? `<span class="fixture-score" data-match="${f.id}">${r.home}:${r.away}</span>`
      : `<span class="fixture-score pending" data-match="${f.id}">vs</span>`;
    return `
      <div class="fixture-row">
        <div class="fixture-teams">
          <span>${homeTeam ? homeTeam.flag + ' ' + homeTeam.name : f.home}</span>
          ${scoreHtml}
          <span>${awayTeam ? awayTeam.flag + ' ' + awayTeam.name : f.away}</span>
        </div>
      </div>`;
  }).join('');

  return `
    <div class="group-card">
      <div class="group-title">Group ${group}</div>
      <table class="standings-table">
        <thead><tr>
          <th>Team</th><th>P</th><th>W</th><th>D</th><th>L</th>
          <th>GF</th><th>GA</th><th>GD</th><th>Pts</th>
        </tr></thead>
        <tbody>${rows}</tbody>
      </table>
      <div class="fixtures-list">${fixtureRows}</div>
    </div>`;
}

```js
// ui/app.js
import { TEAMS } from '../data/teams.js';
import { SCHEDULE } from '../data/schedule.js';
import { predict, teamStrength } from '../engine/predictor.js';
import { groupTable, predictionPanel, teamProfileModal } from './components.js';

const STORAGE_KEY = 'footex_results';
const GROUPS = ['A','B','C','D','E','F','G','H'];

// --- State ---
function loadResults() {
  try { return JSON.parse(localStorage.getItem(STORAGE_KEY)) || {}; }
  catch { return {}; }
}

function saveResult(matchId, home, away) {
  const results = loadResults();
  results[matchId] = { home, away };
  localStorage.setItem(STORAGE_KEY, JSON.stringify(results));
}

function clearResult(matchId) {
  const results = loadResults();
  delete results[matchId];
  localStorage.setItem(STORAGE_KEY, JSON.stringify(results));
}

// --- Views ---
function renderGroups() {
  const results = loadResults();
  return `<div class="groups-grid">
    ${GROUPS.map(g => groupTable(g, TEAMS, SCHEDULE, results)).join('')}
  </div>`;
}

function renderPredict(teamAId = '', teamBId = '') {
  const options = TEAMS
    .slice()
    .sort((a, b) => a.name.localeCompare(b.name))
    .map(t => `<option value="${t.id}" ${t.id === teamAId ? 'selected' : ''}>${t.flag} ${t.name}</option>`)
    .join('');

  const optionsB = TEAMS
    .slice()
    .sort((a, b) => a.name.localeCompare(b.name))
    .map(t => `<option value="${t.id}" ${t.id === teamBId ? 'selected' : ''}>${t.flag} ${t.name}</option>`)
    .join('');

  let resultHtml = `<div class="placeholder-hint">Select two teams to see the prediction</div>`;

  if (teamAId && teamBId && teamAId !== teamBId) {
    const teamA = TEAMS.find(t => t.id === teamAId);
    const teamB = TEAMS.find(t => t.id === teamBId);
    if (teamA && teamB) {
      const result = predict(teamA, teamB, TEAMS);
      resultHtml = predictionPanel(result, teamA, teamB);
    }
  }

  return `
    <div class="predict-container">
      <div class="team-selectors">
        <select class="team-select" id="team-a">
          <option value="">Select Team A...</option>${options}
        </select>
        <div class="vs-label">VS</div>
        <select class="team-select" id="team-b">
          <option value="">Select Team B...</option>${optionsB}
        </select>
      </div>
      ${resultHtml}
    </div>`;
}

// --- Inline edit ---
function startInlineEdit(matchId) {
  const scoreEl = document.querySelector(`[data-match="${matchId}"]`);
  if (!scoreEl) return;

  const results = loadResults();
  const existing = results[matchId];
  const homeVal = existing ? existing.home : '';
  const awayVal = existing ? existing.away : '';

  scoreEl.outerHTML = `
    <span class="inline-edit" data-editing="${matchId}">
      <input class="score-input" id="edit-home" type="number" min="0" max="99" value="${homeVal}" placeholder="0">
      <span>:</span>
      <input class="score-input" id="edit-away" type="number" min="0" max="99" value="${awayVal}" placeholder="0">
      <button class="save-btn" data-save="${matchId}">Save</button>
    </span>`;

  document.getElementById('edit-home').focus();
}

function commitInlineEdit(matchId) {
  const homeInput = document.getElementById('edit-home');
  const awayInput = document.getElementById('edit-away');
  if (!homeInput || !awayInput) return;

  const home = parseInt(homeInput.value, 10);
  const away = parseInt(awayInput.value, 10);

  if (isNaN(home) || isNaN(away)) {
    clearResult(matchId);
  } else {
    saveResult(matchId, home, away);
  }
  renderApp('groups');
}

function cancelInlineEdit() {
  renderApp('groups');
}

// --- Team profile modal ---
function showTeamProfile(teamId) {
  const team = TEAMS.find(t => t.id === teamId);
  if (!team) return;
  const strength = teamStrength(team, TEAMS);
  const upcoming = SCHEDULE.filter(f => (f.home === teamId || f.away === teamId) && !loadResults()[f.id]);
  document.body.insertAdjacentHTML('beforeend', teamProfileModal(team, strength, upcoming, TEAMS));

  document.getElementById('modal-close').addEventListener('click', () => {
    document.getElementById('team-modal').remove();
  });

  document.getElementById('team-modal').addEventListener('click', (e) => {
    if (e.target.id === 'team-modal') document.getElementById('team-modal').remove();
  });
}

// --- Main render ---
let currentView = 'groups';

function renderApp(view) {
  currentView = view || currentView;
  const app = document.getElementById('app');

  if (currentView === 'groups') {
    app.innerHTML = renderGroups();
  } else if (currentView === 'predict') {
    const aId = document.getElementById('team-a')?.value || '';
    const bId = document.getElementById('team-b')?.value || '';
    app.innerHTML = renderPredict(aId, bId);
    wirePredict();
  }
}

function wirePredict() {
  ['team-a', 'team-b'].forEach(id => {
    document.getElementById(id)?.addEventListener('change', () => {
      renderApp('predict');
    });
  });
}

// --- Event delegation ---
document.addEventListener('click', (e) => {
  // Nav links
  const navLink = e.target.closest('[data-view]');
  if (navLink) {
    e.preventDefault();
    document.querySelectorAll('.nav-link').forEach(l => l.classList.remove('active'));
    navLink.classList.add('active');
    renderApp(navLink.dataset.view);
    return;
  }

  // Fixture score click → start inline edit
  const scoreEl = e.target.closest('.fixture-score');
  if (scoreEl) {
    startInlineEdit(scoreEl.dataset.match);
    return;
  }

  // Save button in inline edit
  const saveBtn = e.target.closest('[data-save]');
  if (saveBtn) {
    commitInlineEdit(saveBtn.dataset.save);
    return;
  }

  // Team name → profile modal
  const teamBtn = e.target.closest('[data-team]');
  if (teamBtn) {
    showTeamProfile(teamBtn.dataset.team);
    return;
  }
});

// Escape cancels inline edit
document.addEventListener('keydown', (e) => {
  if (e.key === 'Escape' && document.querySelector('[data-editing]')) cancelInlineEdit();
  if (e.key === 'Enter' && document.querySelector('[data-editing]')) {
    const editing = document.querySelector('[data-editing]');
    if (editing) commitInlineEdit(editing.dataset.editing);
  }
});

// Boot
renderApp('groups');
```
